General Rule:

selector{
	property: value;
	anotherProperty: value;
}

example
//Make all h1's purple with font size of 56px
h1{
	color: purple;
	font-size: 56px;
}

//Make all img's red borders that are 3px thick;
img{
	border-color: red;
	border-width: 3px;
}
